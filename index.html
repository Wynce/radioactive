        function updateOverallProgressPercentage() {

        function checkSectionCompletion(sectionElement) {

            if (!sectionElement || !sectionElement.classList.contains('topic-section')) return;

            const inputsInSection = sectionElement.querySelectorAll('.blank-input');

            let allCorrectInSection = true;

            if (inputsInSection.length > 0) {

                for (let input of inputsInSection) {

                    const statusBox = document.getElementById(`status-${input.id}`);

                    if (!statusBox || !(statusBox.classList.contains('correct') || statusBox.classList.contains('revealed'))) {

                        allCorrectInSection = false;

                        break;

                    }

                }

            } else {

                allCorrectInSection = false;

            }

            const sectionIndex = topicSectionsDOMElements.indexOf(sectionElement);

            if (sectionIndex === -1) return; // Section not found in our tracked array

            if (allCorrectInSection) {

                sectionElement.classList.add('all-correct');

                    progressSegmentsDOMElements[sectionIndex].classList.add('defeated');

                }

            } else {

                sectionElement.classList.remove('all-correct');

                    progressSegmentsDOMElements[sectionIndex].classList.remove('defeated');

                    progressSegmentsDOMElements[sectionIndex].classList.remove('crushed-effect'); // UNCOMMENTED

                }

            }

            updateOverallProgressPercentage();

        }

                progressSegmentsDOMElements.push(segment);

            });

            updateOverallProgressPercentage();

        }

        function structureTopics(content) {

            if (!notesContainer) return;

            const tempDiv = document.createElement('div');

            tempDiv.innerHTML = content;

            notesContainer.innerHTML = '';

            // topicSectionsDOMElements = []; // This was reset too early, now populated after DOM manipulation

            let currentSection = null;

            Array.from(tempDiv.childNodes).forEach(node => {

                if (node.nodeName === 'H2') {

                    if (currentSection) notesContainer.appendChild(currentSection); // Append previous section

                    currentSection = document.createElement('div');

                    currentSection.className = 'topic-section';

                    currentSection.appendChild(node.cloneNode(true));

                    // No push to topicSectionsDOMElements here yet, will query later

                } else if (currentSection) { // Make sure currentSection exists before appending children

                    if (node.nodeName === 'STRONG') {

                        currentSection.appendChild(node.cloneNode(true));

                    } else if (node.nodeType === Node.TEXT_NODE && node.textContent.trim() !== '') {

                        const p = document.createElement('p'); p.className = 'interactive-paragraph'; p.textContent = node.textContent.trim(); currentSection.appendChild(p);

                    } else if (node.nodeType === Node.ELEMENT_NODE) {

                        if (node.classList && node.classList.contains('sentence-container')) {

                            const p = document.createElement('p'); p.className = 'interactive-paragraph'; p.appendChild(node.cloneNode(true)); currentSection.appendChild(p);

                        } else if (node.tagName === 'TABLE') {

                            const w = document.createElement('div'); w.className = 'table-responsive-wrapper'; w.appendChild(node.cloneNode(true)); currentSection.appendChild(w);

                        } else if (node.tagName === 'OL' || node.tagName === 'UL') {

                            currentSection.appendChild(node.cloneNode(true));

                        } else if (node.textContent.trim() !== '') {

                             const p = document.createElement('p'); p.className = 'interactive-paragraph'; p.innerHTML = node.innerHTML || node.textContent; currentSection.appendChild(p);

                        }

                    }

                } else { // Content before first H2 (should not be topic sections)

                    if (node.nodeType === Node.TEXT_NODE && node.textContent.trim() !== '') {

                        const p = document.createElement('p'); p.className = 'interactive-paragraph'; p.textContent = node.textContent.trim(); notesContainer.appendChild(p);

                    } else if (node.nodeType === Node.ELEMENT_NODE) {

                        notesContainer.appendChild(node.cloneNode(true)); // Append non-section elements directly

                    }

                }

            });

            // **Crucial Fix:** Append the very last section if it exists

            if (currentSection) {

                notesContainer.appendChild(currentSection);

            }

           

            // Now that all sections are in the DOM, query them

            topicSectionsDOMElements = Array.from(notesContainer.querySelectorAll('.topic-section'));

            initializeSegmentedProgressBar();

        }

        function createInteractiveNotes() {

            if (!notesContainer) return;

            structureTopics(notesContent);

            blankCounter = 0;

            notesContainer.querySelectorAll('span.sentence-container').forEach(el => {

                if (!el || !el.innerHTML || !el.innerHTML.includes('[')) return;

                let originalHTML = el.innerHTML;

                let hasChanged = false;

                const processedHTML = originalHTML.replace(/\[([^\]]+)\]/g, (match, word) => {

                    hasChanged = true;

                    const firstLetter = word[0];

                    const wordsInAnswer = word.split(' ');

                    let placeholderHint = wordsInAnswer.map(w => w[0] + '_'.repeat(Math.max(0,w.length-1))).join(' ');

                    if (wordsInAnswer.length === 1) placeholderHint = firstLetter + '_'.repeat(Math.max(0, word.length - 1));

                    const inputId = `blank-${blankCounter++}`;

                    const inputSize = Math.min(Math.max(word.length, 8), 25);

                    return `

                        <span class="blank-container">

                            <input type="text" class="blank-input" id="${inputId}"

                                   placeholder="${placeholderHint}"

                                   data-answer="${word.toLowerCase()}"

                                   data-original-answer="${word}"

                                   style="width: ${inputSize * 0.8}em;"

                                   autocomplete="off" autocorrect="off" autocapitalize="off" spellcheck="false">

                            <span class="status-box pending" id="status-${inputId}"></span>

                            <button class="reveal-answer-btn" data-input-id="${inputId}" title="Show Answer">👁️</button>

                        </span>

                    `;

                });

                if(hasChanged) el.innerHTML = processedHTML;

            });

            document.querySelectorAll('.blank-input').forEach(input => {

                input.setAttribute('data-initial-placeholder', input.placeholder);

                input.addEventListener('input', function() {

                    const statusBox = document.getElementById(`status-${this.id}`);

                    const parentSection = this.closest('.topic-section');

                    const answer = this.dataset.answer;

                    const userInput = this.value.trim().toLowerCase();

                    const isCorrectByInput = userInput === answer ||

                                      userInput === (answer.endsWith('s') ? answer.slice(0, -1) : answer + 's') ||

                                      (answer.includes(' ') && userInput === answer.replace(/\s+/g, ''));

                    let typedSoFar = this.value;

                    if (typedSoFar.length === 0) this.placeholder = this.getAttribute('data-initial-placeholder');

                    else {

                        const originalAnswer = this.dataset.originalAnswer;

                        this.placeholder = typedSoFar + (originalAnswer.length > typedSoFar.length ? '_'.repeat(originalAnswer.length - typedSoFar.length) : '');

                    }

                    if (isCorrectByInput) {

                        statusBox.className = 'status-box correct';

                        this.style.borderBottomColor = 'var(--secondary-color)';

                    } else if (userInput !== '') {

                        statusBox.className = 'status-box incorrect';

                        this.style.borderBottomColor = 'var(--error-color)';

                    } else {

                        statusBox.className = 'status-box pending';

                        this.style.borderBottomColor = 'var(--dark-gray)';

                    }

                    if (parentSection) checkSectionCompletion(parentSection);

                });

            });

            document.querySelectorAll('.reveal-answer-btn').forEach(btn => {

                btn.addEventListener('click', function() {

                    if (this.classList.contains('used')) return;

                    const inputId = this.dataset.inputId;

                    const inputField = document.getElementById(inputId);

                    const statusBox = document.getElementById(`status-${inputId}`);

                    const parentSection = inputField.closest('.topic-section');

                    inputField.value = inputField.dataset.originalAnswer;

                    inputField.readOnly = true;

                    inputField.classList.add('revealed');

                    this.classList.add('used');

                    this.innerHTML = '✔️';

                    statusBox.className = 'status-box revealed';

                    if (parentSection) checkSectionCompletion(parentSection);

                });

            });

            topicSectionsDOMElements.forEach(checkSectionCompletion);

        }

        document.addEventListener('DOMContentLoaded', () => {

            try {

                createInteractiveNotes();

            } catch (error) {

                console.error("Error during createInteractiveNotes():", error);

            }

        });

    </script>
